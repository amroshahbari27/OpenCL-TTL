19/*The Khronos TTL runtime library (event_t, tiler, DMA helpers) is only
20 * needed when compiling for actual OpenCL execution. The TTL compiler
21 * frontend (cgeist) only needs the scheduling macros below --- pass
22 * -DTTL_MACROS_ONLY on the cgeist command line to skip the library. */
23
24/*===============================================================================
25 * TTL Compiler Macros
26 *
27 * Scheduling hints consumed by the TTL compiler (cgeist -x cl + ttl-opt).
28 * Standard OpenCL compilers ignore unknown pragmas --- safe in any source.
29 * No tiling? Just write normal for-loops --- the compiler won't touch them.
30 *
31 * Usage:
32 * #include "TTL/TTL.h"
33 *
34 * __kernel void gemm(const TTL_TENSOR_2D(A, int, M, K),
35 * const TTL_TENSOR_2D(B, int, K, N),
36 * TTL_TENSOR_2D(C, int, M, N)) {
37 *
38 * TTL_LOOP_3D(i, 0, M, 64,
39 * j, 0, N, 64,
40 * k, 0, K, 64,
41 * TTL_DOUBLE_BUFFER, {
42 * TTL_2D(C, i,1,0, j,1,0) += TTL_2D(A, i,1,0, k,1,0) * TTL_2D(B, k,1,0, j,1,0);
43 * })
44 * }
45 *
46 * Body is a macro argument (trailing { ... }, not a bare brace block after
47 * the call) -- same syntax on both the C and C++ paths. IV name hiding:
48 * TTL_LOOP declares 'i_MUST_BE_DECLARED_BY_TTL_LOOP' (not 'i'), and
49 * TTL_2D(C, i, ...) expands to reference that exact derived name (C mode)
50 * or a body-local '_TTL_Induc'-typed alias named 'i' the macro declares
51 * from it (C++ mode, see the C++ branch below for why). If 'i' didn't
52 * come from TTL_LOOP, the reference fails to compile either way.
53 *
54 * Trade-off: the error message for misuse is "undeclared identifier 'x_iv_ttl'"
55 * rather than a domain-specific diagnostic. Also, the user cannot reference
56 * the loop IV directly in the body (e.g. 'int row = i * width' fails because
57 * 'i' doesn't exist -- the actual variable is 'i_iv_ttl'). This is the cost
58 * of structural enforcement via C macros; a cgeist-level diagnostic could
59 * provide better error messages in the future.
60 * =============================================================================== */
61
62/*===============================================================================
63 * Kernel Parameter Macros
64 *
65 * TTL_TENSOR_2D(A, int, M, K) -> __global int _ttl_A[restrict M][K]
66 * Access via TTL_2D(A, i,1,0, k,1,0) expands to _ttl_A[1*i+0][1*k+0]
67 *
68 * The raw variable is hidden --- all accesses must go through TTL().
69 *
70 * LIMITATION: VLA declarations assume contiguous row-major storage
71 * (stride == innermost dimension). Unlike the Khronos runtime, which
72 * separates TTL_shape_t (logical size) from TTL_layout_t (memory stride),
73 * our macros cannot represent padded allocations where stride > width
74 * or sub-views into a larger matrix. This is a deliberate trade-off:
75 * full-rank static shapes give MLIR complete affine analysis information,
76 * at the cost of not supporting non-contiguous global layouts.
77 * Future work: an optional stride parameter could lift this restriction.
78 * =============================================================================== */
79
80/*-----------------------------------------------------------------------
81 * DESIGN REQUIREMENT: Single code path
82 *
83 * The SAME source file must compile and run correctly via:
84 * 1. cgeist → ttl-opt → ttl-translate (compiler optimization path)
85 * 2. OpenCL compiler (e.g. PoCL) directly (unoptimized execution)
86 *
87 * No #ifdef splits between the two paths. Every macro must expand to
88 * valid C / OpenCL C in both contexts.
89 *
90 * RESOLVED: Compile-time constant enforcement (scale/offset)
91 *
92 * The scale and offset fields in TTL_1D/2D/3D (si, oi, sj, oj, ...)
93 * MUST be compile-time constants for correct affine analysis. This used
94 * to be enforced ONLY by the compiler downstream (cgeist rejects
95 * non-affine patterns with a confusing error far from the macro
96 * expansion site) -- now it's enforced right at the TTL_1D/2D/3D call
97 * site, with a real, specific diagnostic.
98 *
99 * _Static_assert doesn't work here (it's a declaration, not an
100 * expression -- can't appear inside TTL_1D/2D/3D's array subscripts),
101 * and the classic __builtin_constant_p + sizeof(int[cond?1:-1]) trick,
102 * while it DOES work (verified directly against cgeist -- Clang's Sema
103 * checks array-size validity, and Polygeist reuses Sema unmodified),
104 * only gives a generic "array size is negative" error with no domain
105 * information. A second attempt, __attribute__((error(...))) on a
106 * poisoned function gated by __builtin_constant_p in a ternary, gives a
107 * fully custom message on plain clang -- but does NOT survive cgeist:
108 * verified directly, Polygeist's custom AST-to-MLIR walker doesn't
109 * replicate the CodeGen-level dead-branch elimination this trick needs,
110 * so it conservatively assumes __builtin_constant_p is always false and
111 * calls the poisoned function unconditionally, breaking even the
112 * correct/constant case.
113 *
114 * WORKING: __attribute__((diagnose_if(...))) (Clang-specific, checked
115 * during Sema -- same phase Polygeist reuses unmodified, which is why
116 * this survives cgeist where the poison-function trick didn't). Verified
117 * directly: the constant case compiles clean and inlines away completely
118 * (identical affine.load/store MLIR to the unchecked version, zero
119 * codegen cost); the non-constant case fails with the exact custom
120 * message below, at the exact usage site, on both cgeist and the
121 * direct-exec leg (clang -cl-std=... -> SPIR-V). Both TTLCompiler legs
122 * are Clang-based, so diagnose_if's non-portability to GCC isn't a
123 * practical restriction here.
124 *
125 * IV provenance (as opposed to scale/offset constant-ness) is a
126 * different, harder problem -- see the C++ branch below and
127 * TTL-CPP-STRUCTURED-ACCESS-DESIGN.md §16 for why no C-portable
128 * equivalent of the C++ _TTL_Induc body-alias check was found; plain C
129 * keeps the name-mangled rename (_TTL_IV, "hide it behind a prefix/suffix
130 * so the wrong name doesn't exist") as its only IV-provenance mechanism.
131 * Every angle tried to give it C++'s type-based protection instead was
132 * ruled out directly, not assumed:
133 * - plain C `enum` gives zero protection on its own -- implicit
134 * int-to-enum conversion is silently allowed, even at -Wall -Wextra.
135 * - Clang DOES have a warning for exactly this
136 * (-Wimplicit-int-enum-cast, promotable to a hard error via a
137 * _Pragma right in this header) -- but it does not exist in Clang 18,
138 * the actual pinned toolchain both TTLCompiler legs use (confirmed:
139 * grepped a -Weverything run against the real pinned binary). It
140 * only appears in much newer Clang. So this specific fix is blocked
141 * on a toolchain upgrade, not a fundamental language limitation --
142 * if this project ever moves off Clang 18, this is worth retrying.
143 * - The one diagnostic Clang 18 does have for this
144 * (-Wsign-conversion) is too broad to promote to an error: it fires
145 * 129 times on TTL's own existing upstream runtime headers alone.
146 * - C11 _Generic (exact-type dispatch, which would sidestep the
147 * implicit-conversion problem entirely) crashes cgeist unconditionally
148 * -- confirmed with the simplest possible _Generic expression, no
149 * relation to IV-checking specifically. Polygeist's MLIRScanner has
150 * no visitor for GenericSelectionExpr at all.
151 * ----------------------------------------------------------------------- */

153/*Compile-time-constant checks for TTL_1D/2D/3D's scale/offset. Each
154 * function just returns its argument unchanged (fully inlined away by
155 * cgeist/clang for any real, constant call -- verified: zero codegen
156 * cost, affine recognition unaffected) -- diagnose_if is what actually
157 * does the work, firing only when the argument isn't a compile-time
158 * constant. */
159static inline int _ttl_check_constant_scale(int scale)
160 __attribute__((diagnose_if(!__builtin_constant_p(scale), "TTL_1D/2D/3D: scale must be a compile-time constant",
161 "error"))) {
162 return scale;
163}
164static inline int _ttl_check_constant_offset(int offset)
165 __attribute__((diagnose_if(!__builtin_constant_p(offset), "TTL_1D/2D/3D: offset must be a compile-time constant",
166 "error"))) {
167 return offset;
168}
